using UnityEngine; using UnityEngine.UI; public class GameController : MonoBehaviour { [Header("Movement")] public float speed = 5f; public float mouseSensitivity = 200f; public Transform playerBody; float xRotation = 0f; [Header("Time Control")] public float timeEnergy = 5f; public float maxEnergy = 5f; public float drainSpeed = 1f; public float rechargeSpeed = 0.5f; private bool isTimeStopped = false; [Header("Health")] public int health = 100; [Header("Enemy")] public Transform enemy; public float enemySpeed = 3f; [Header("UI")] public Slider energyBar; public Slider healthBar; void Start() { Cursor.lockState = CursorLockMode.Locked; if (energyBar != null) energyBar.maxValue = maxEnergy; if (healthBar != null) healthBar.maxValue = health; } void Update() { Move(); Look(); HandleTime(); RechargeEnergy(); UpdateUI(); EnemyAI(); } // 🎮 الحركة void Move() { float x = Input.GetAxis("Horizontal"); float z = Input.GetAxis("Vertical"); Vector3 move = playerBody.right * x + playerBody.forward * z; transform.Translate(move * speed * Time.deltaTime, Space.World); } // 👀 الكاميرا void Look() { float mouseX = Input.GetAxis("Mouse X") * mouseSensitivity * Time.deltaTime; float mouseY = Input.GetAxis("Mouse Y") * mouseSensitivity * Time.deltaTime; xRotation -= mouseY; xRotation = Mathf.Clamp(xRotation, -90f, 90f); Camera.main.transform.localRotation = Quaternion.Euler(xRotation, 0f, 0f); playerBody.Rotate(Vector3.up * mouseX); } // ⏳ التحكم بالزمن void HandleTime() { if (Input.GetKey(KeyCode.Space) && timeEnergy > 0) { Time.timeScale = 0f; isTimeStopped = true; timeEnergy -= drainSpeed * Time.unscaledDeltaTime; } else { Time.timeScale = 1f; isTimeStopped = false; } } void RechargeEnergy() { if (!isTimeStopped && timeEnergy < maxEnergy) { timeEnergy += rechargeSpeed * Time.deltaTime; } } // 👾 العدو void EnemyAI() { if (enemy == null) return; if (Time.timeScale > 0f) { enemy.position = Vector3.MoveTowards( enemy.position, transform.position, enemySpeed * Time.deltaTime ); float distance = Vector3.Distance(enemy.position, transform.position); if (distance < 2f) { TakeDamage(1); } } } // ❤️ الصحة void TakeDamage(int damage) { health -= damage; if (health <= 0) { Debug.Log("Game Over"); Time.timeScale = 0f; } } // 🎨 UI void UpdateUI() { if (energyBar != null) energyBar.value = timeEnergy; if (healthBar != null) healthBar.value = health; } }
